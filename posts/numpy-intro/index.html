An example

import numpy as np

a = np.arange(15).reshape(3,5)

print(a)
print(a.ndim)
print(a.shape)
print(a.size)
print(a.dtype)
print(a.itemsize)
print(a.data)

Array Creation

1-D array

# create an array from a regular python list or tuple
a = np.array([2,3,4])
b = np.array((5,6,7)) # tuple
c = np.array((8.1,9.2,10.3))
print(a)
print(a.dtype)
print(a.itemsize)

print(b)
print(b.dtype)
print(b.itemsize)
